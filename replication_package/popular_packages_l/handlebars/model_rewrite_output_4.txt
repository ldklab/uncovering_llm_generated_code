```javascript
// index.js
class BasicHandlebars {
  constructor() {
    this.helpers = {};
  }

  // Compiles the template into a rendering function
  compile(template) {
    return (context) => this.render(template, context);
  }

  // Renders the template by replacing placeholders with context data
  render(template, context) {
    return template.replace(/\{\{(\w+)\}\}/g, (match, prop) => this.resolvePath(context, prop));
  }

  // Resolves the value of the given property from the context
  resolvePath(context, prop) {
    return context[prop] ?? '';
  }

  // Registers a helper function that can transform sections of the template
  registerHelper(name, fn) {
    this.helpers[name] = fn;
  }

  // Applies registered helpers to the template
  applyHelpers(template, context) {
    return template.replace(/\{\{#(\w+) (.+?)\}\}(.*?)\{\{\/\1\}\}/gs, (match, helper, args, inner) => {
      if (this.helpers[helper]) {
        return this.helpers[helper](JSON.parse(args), inner, context);
      }
      return match;
    });
  }
}

module.exports = BasicHandlebars;
```

```javascript
// test.js
const BasicHandlebars = require('./index');

// Template with placeholders
const source = "<p>Hello, my name is {{name}}. I am from {{hometown}}.</p>";

// Sample context data
const data = { name: "Alan", hometown: "Somewhere, TX" };

// Create a new instance of BasicHandlebars
const handlebars = new BasicHandlebars();

// Compile the template into a render function
const template = handlebars.compile(source);

// Render the template with the given context data
const result = template(data);

console.log(result); 
// Expected Output:
// <p>Hello, my name is Alan. I am from Somewhere, TX.</p>
```
