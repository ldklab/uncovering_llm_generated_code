// File: build.js
const fs = require('fs');
const path = require('path');

function build() {
  const srcPath = path.resolve(__dirname, 'src');
  const distPath = path.resolve(__dirname, 'dist');

  if (!fs.existsSync(distPath)){
    fs.mkdirSync(distPath, { recursive: true });
  }
  
  const submodulesSrcPath = path.join(srcPath, 'submodules');
  const submodules = fs.readdirSync(submodulesSrcPath);

  submodules.forEach(module => {
    const moduleSrc = path.join(submodulesSrcPath, module);
    const moduleDist = path.join(distPath, 'submodules', module);

    if (!fs.existsSync(moduleDist)){
      fs.mkdirSync(moduleDist, { recursive: true });
    }

    fs.copyFileSync(path.join(moduleSrc, 'index.ts'), path.join(moduleDist, 'index.js'));
  });
}

build();
