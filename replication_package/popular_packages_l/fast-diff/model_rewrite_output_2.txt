The `fast-diff.js` Node.js module provides a function to compute the difference between two strings, `text1` and `text2`. The module identifies the differing sections between the two texts and returns an array consisting of tuples. Each tuple contains a numerical flag indicating whether the text is `"equal"`, `"inserted"`, or `"deleted"`, and the corresponding string segment.

### Explanation:
- **Constants:** The module defines constants `DIFF_DELETE`, `DIFF_INSERT`, and `DIFF_EQUAL` with values `-1`, `1`, and `0` respectively. These represent the types of operations on text (deletion, insertion, equality).
  
- **Exported Functionality:**
  - `diff(text1, text2, cursorPos)`: This is the main function that calculates the difference between the two given texts.
  
- **Utility Functions:**
  - `commonPrefix(text1, text2)`: Determines the length of the common prefix shared by `text1` and `text2`.
  - `commonSuffix(text1, text2)`: Determines the length of the common suffix shared by `text1` and `text2`.
  - `computeDiff(text1, text2)`: Core function to compute the internal differences after removing common prefixes and suffixes.
  - `findMiddleSnake(text1, text2)`: Placeholder for an internal algorithm to find a "middle snake" in the O(ND) difference algorithm, assisting in segmenting the texts for recursive diff computation.

### Function Flow:
1. If `text1` is equal to `text2`, a single `DIFF_EQUAL` operation is returned.
2. The longest common prefix and suffix are identified and stripped to focus on the differing middle sections.
3. `computeDiff` is called to identify differences in the core differing section.
4. Common prefix and suffix are attached back to indicate unchanged parts.

```javascript
// fast-diff.js

const DIFF_DELETE = -1;
const DIFF_INSERT = 1;
const DIFF_EQUAL = 0;

module.exports = function diff(text1, text2, cursorPos) {
  if (text1 === text2) {
    return text1 ? [[DIFF_EQUAL, text1]] : [];
  }

  const commonPrefixLength = commonPrefix(text1, text2);
  const commonSuffixLength = commonSuffix(text1.slice(commonPrefixLength), text2.slice(commonPrefixLength));

  text1 = text1.slice(commonPrefixLength, text1.length - commonSuffixLength);
  text2 = text2.slice(commonPrefixLength, text2.length - commonSuffixLength);

  let diffs = computeDiff(text1, text2);

  if (commonPrefixLength) {
    diffs.unshift([DIFF_EQUAL, text1.slice(0, commonPrefixLength)]);
  }
  if (commonSuffixLength) {
    diffs.push([DIFF_EQUAL, text1.slice(-commonSuffixLength)]);
  }

  return diffs;
};

module.exports.INSERT = DIFF_INSERT;
module.exports.EQUAL = DIFF_EQUAL;
module.exports.DELETE = DIFF_DELETE;

function commonPrefix(text1, text2) {
  const n = Math.min(text1.length, text2.length);
  for (let i = 0; i < n; i++) {
    if (text1[i] !== text2[i]) return i;
  }
  return n;
}

function commonSuffix(text1, text2) {
  const text1Length = text1.length;
  const text2Length = text2.length;
  const n = Math.min(text1Length, text2Length);
  for (let i = 0; i < n; i++) {
    if (text1[text1Length - i - 1] !== text2[text2Length - i - 1]) return i;
  }
  return n;
}

function computeDiff(text1, text2) {
  const diffs = [];

  if (!text1) {
    return [[DIFF_INSERT, text2]];
  }
  if (!text2) {
    return [[DIFF_DELETE, text1]];
  }

  const middlePoint = findMiddleSnake(text1, text2);

  if (middlePoint === -1) {
    diffs.push([DIFF_DELETE, text1]);
    diffs.push([DIFF_INSERT, text2]);
  } else {
    const [x, y] = middlePoint;

    const leftPartDiffs = computeDiff(text1.slice(0, x), text2.slice(0, y));
    const rightPartDiffs = computeDiff(text1.slice(x), text2.slice(y));

    diffs = diffs.concat(leftPartDiffs, [[DIFF_EQUAL, text1[x]]], rightPartDiffs);
  }

  return diffs;
}

function findMiddleSnake(text1, text2) {
  // Placeholder for detailed implementation
  return [Math.floor(text1.length / 2), Math.floor(text2.length / 2)];
}
```
