// index.js
function identifyBoxedPrimitive(value) {
  if (value === null || typeof value !== 'object') {
    return null;
  }
  
  const typeCheck = Object.prototype.toString.call(value);
  
  switch (typeCheck) {
    case '[object Boolean]':
      return 'Boolean';
    case '[object Number]':
      return 'Number';
    case '[object String]':
      return 'String';
    case '[object Symbol]':
      return 'Symbol';
    case '[object BigInt]':
      return 'BigInt';
    default:
      return undefined;
  }
}

module.exports = identifyBoxedPrimitive;
