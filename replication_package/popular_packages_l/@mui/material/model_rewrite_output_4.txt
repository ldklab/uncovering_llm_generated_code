```javascript
// install-material-ui.js
const { exec } = require('child_process');

const command = 'npm install @mui/material @emotion/react @emotion/styled';

exec(command, (error, stdout, stderr) => {
  if (error) {
    console.error(`Error installing Material UI: ${error.message}`);
    return;
  }
  if (stderr) {
    console.error(`Installation stderr: ${stderr}`);
    return;
  }
  console.log(`Installation stdout: ${stdout}`);
});
```

```javascript
// open-docs.js
const open = require('open');
const url = 'https://mui.com/material-ui/';

open(url)
  .then(() => {
    console.log('Material UI documentation opened in browser.');
  })
  .catch((error) => {
    console.error(`Error opening documentation: ${error.message}`);
  });
```

The provided Node.js code defines a simple custom manager for Material UI. The manager consists of a package.json file with scripts to install Material UI and open its documentation. The `install-material-ui.js` script installs Material UI and its dependencies using npm, and logs any errors or outputs. The `open-docs.js` script opens the Material UI documentation website using the `open` package.
