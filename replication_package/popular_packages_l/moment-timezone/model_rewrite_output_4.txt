```javascript
// index.js
const moment = require('moment-timezone');

function displayTimeInZones(dateInput) {
  const date = moment(dateInput);
  const timeZones = [
    'America/Los_Angeles',
    'America/New_York',
    'Asia/Tokyo',
    'Australia/Sydney'
  ];

  timeZones.forEach(timeZone => {
    console.log(`${timeZone}: ${date.tz(timeZone).format('ha z')}`);
  });
}

console.log('Displaying times for June 1, 2014 12:00 UTC');
displayTimeInZones("2014-06-01T12:00:00Z");

console.log('Displaying times for December 1, 2014 12:00 UTC');
displayTimeInZones("2014-12-01T12:00:00Z");
```
