const babelParser = require('@babel/parser');

function parseJavaScriptCode(code) {
  try {
    // Parse JavaScript code into an Abstract Syntax Tree (AST)
    const ast = babelParser.parse(code, {
      sourceType: 'module',
      plugins: ['jsx', 'flow']
