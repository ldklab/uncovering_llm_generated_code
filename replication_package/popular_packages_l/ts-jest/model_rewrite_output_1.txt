```javascript
// ts-jest-transformer.js
const ts = require('typescript');

module.exports = {
  process(src, path) {
    if (path.endsWith('.ts')) {
      const transpileOptions = {
        compilerOptions: {
          module: ts.ModuleKind.CommonJS,
          sourceMap: true
        }
      };
      const { outputText } = ts.transpileModule(src, transpileOptions);
      return outputText;
    }
    return src;
  }
};
```
