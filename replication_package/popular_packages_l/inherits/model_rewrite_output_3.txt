// inherits_browser.js (Browser implementation with ES6 syntax)
module.exports = function inherits(subClass, superClass) {
  if (superClass) {
    subClass.super_ = superClass;
    subClass.prototype = Object.create(superClass.prototype, {
      constructor: {
        value: subClass,
        enumerable: false,
        writable: true,
        configurable: true
      }
    });
  }
};
