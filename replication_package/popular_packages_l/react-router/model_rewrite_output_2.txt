class Router {
  constructor(routes) {
    this.routes = routes;
    this.currentPath = window.location.pathname;
  }

  init() {
    window.addEventListener('popstate', () => {
      this.updatePath(window.location.pathname);
    });
    this.render();
  }

  navigate(path) {
    window.history.pushState({}, '', path);
    this.updatePath(path);
  }

  updatePath(path) {
    this.currentPath = path;
    this.render();
  }

  getViewForCurrentPath() {
    const route = this.routes.find(route => route.path === this.currentPath);
    return route ? route.view : this.routes.find(route => route.path === '*').view;
  }

  render() {
    const view = this.getViewForCurrentPath();
    document.getElementById('app').innerHTML = view();
  }
}

const routes = [
  {
    path: '/',
    view: () => '<h1>Home</h1>',
  },
  {
    path: '/about',
    view: () => '<h1>About</h1>',
  },
  {
    path: '*',
    view: () => '<h1>404 Not Found</h1>',
  },
];

const router = new Router(routes);
router.init();

function navigateTo(path) {
  router.navigate(path);
}

document.getElementById('to-about').addEventListener('click', () => navigateTo('/about'));
document.getElementById('to-home').addEventListener('click', () => navigateTo('/'));
