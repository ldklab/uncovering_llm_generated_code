// index.js
class Router {
  constructor(routes) {
    this.routes = routes;
    this.currentPath = window.location.pathname;
  }

  init() {
    window.onpopstate = this.updateView.bind(this);
    this.updateView();
  }

  navigate(path) {
    window.history.pushState({}, path, window.location.origin + path);
    this.currentPath = path;
    this.updateView();
  }

  updateView() {
    const matchingRoute = this.routes.find(route => route.path === this.currentPath);
    const view = matchingRoute ? matchingRoute.view() : this.routes.find(route => route.path === '*').view();
    this.render(view);
  }

  render(viewContent) {
    document.getElementById('app').innerHTML = viewContent;
  }
}

// Example routes configuration
const routes = [
  {
    path: '/',
    view: () => '<h1>Home</h1>',
  },
  {
    path: '/about',
    view: () => '<h1>About</h1>',
  },
  {
    path: '*',
    view: () => '<h1>404 Not Found</h1>',
  },
];

// Initialize and start the router instance
const router = new Router(routes);
router.init();

// Functions for navigation
function navigateTo(path) {
  router.navigate(path);
}

// Example usage for navigation controls
document.getElementById('to-about').addEventListener('click', () => navigateTo('/about'));
document.getElementById('to-home').addEventListener('click', () => navigateTo('/'));
