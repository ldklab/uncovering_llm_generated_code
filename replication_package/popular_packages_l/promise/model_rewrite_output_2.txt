    this.result = undefined;
    this.callbacks = [];

    const resolve = (value) => {
      if (this.state !== 'pending') return;
      this.state = 'fulfilled';
      this.result = value;

    } catch (error) {
      reject(error);
    }

      callback.onRejected(this.result);
    }
