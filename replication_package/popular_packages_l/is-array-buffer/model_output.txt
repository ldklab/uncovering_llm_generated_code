The `is-array-buffer` package provides a utility function to determine if a given value is a JavaScript `ArrayBuffer`. This detection accounts for the complexities introduced by cross-realm/iframe contexts, does not rely on `instanceof` checks, and handles cases where the `Symbol.toStringTag` might be tampered with. This makes the function robust and reliable for accurately identifying whether a value is an `ArrayBuffer`.

Here's how you can use this package:
- It exports a single function, `isArrayBuffer`, which takes one argument.
- The function returns `true` if the argument is an `ArrayBuffer`, and `false` otherwise.
- It correctly handles cases for various types of inputs, including primitives, objects, functions, and instances of `ArrayBuffer` or its subclasses.

Below is the Node.js implementation of this functionality.

```javascript
// is-array-buffer.js

'use strict';

function isArrayBuffer(value) {
  // Check for null or undefined
  if (value == null) {
    return false;
  }

  // Check the type using Object.prototype.toString
  return Object.prototype.toString.call(value) === '[object ArrayBuffer]';
}

module.exports = isArrayBuffer;

// test.js

var assert = require('assert');
var isArrayBuffer = require('./is-array-buffer');

// Test cases
assert(!isArrayBuffer(function () {}));
assert(!isArrayBuffer(null));
assert(!isArrayBuffer(function* () { yield 42; return Infinity; }));
assert(!isArrayBuffer(Symbol('foo')));
assert(!isArrayBuffer(1n));
assert(!isArrayBuffer(Object(1n)));

assert(!isArrayBuffer(new Set()));
assert(!isArrayBuffer(new WeakSet()));
assert(!isArrayBuffer(new Map()));
assert(!isArrayBuffer(new WeakMap()));
assert(!isArrayBuffer(new WeakRef({})));
assert(!isArrayBuffer(new FinalizationRegistry(() => {})));
assert(!isArrayBuffer(new SharedArrayBuffer()));

assert(isArrayBuffer(new ArrayBuffer()));

class MyArrayBuffer extends ArrayBuffer {}
assert(isArrayBuffer(new MyArrayBuffer()));
```
