The `call-bind` Node.js package provides an utility to bind the `call` method robustly to functions, allowing you to use the `function.call(context, args...)` syntax reliably, even if the `call` method of `Function.prototype` has been modified or deleted. The package exports two main features:

1. **`callBind(fn)`:** This function binds the `.call` method to a specific function `fn`. This allows you to enforce `fn` to always be called with its desired `this` context, and with the arguments you provide.

2. **`callBound(method)`:** This function allows you to extract and bind methods (by their full path, e.g., `Array.prototype.slice`) to ensure they work even if their original prototypes are modified.

### Key Features
- The package helps maintain consistent behavior of JavaScript functions by providing a bound version of `Function.prototype.call`.
- The `callBound()` utility enables using JavaScript methods such as `Array.prototype.slice` without being affected by modification or deletion of the method from its prototype.

Below is the complete code for the `call-bind` Node.js package that implements these functionalities:

```markdown
// call-bind/index.js
'use strict';

/**
 * Bind `Function.prototype.call` to a given function `fn`.
 * @param {Function} fn - The function to bind.
 * @returns {Function} - A function with bound `call`.
 */
module.exports = function callBind(fn) {
  return function boundFunction() {
    return Function.prototype.call.apply(fn, arguments);
  };
};

// call-bind/callBound.js
'use strict';

/**
 * Retrieve a method from an object and bind it, ensuring it works
 * even if the original prototype method is modified or deleted.
 * @param {string} methodPath - The path to the method (e.g., 'Array.prototype.slice').
 * @returns {Function} - A bound function.
 */
module.exports = function callBound(methodPath) {
  const pathParts = methodPath.split('.');
  let obj = globalThis;

  for (let i = 0; i < pathParts.length; i++) {
    obj = obj[pathParts[i]];
  }

  if (typeof obj !== 'function') {
    throw new TypeError('path must resolve to a function');
  }

  return Function.prototype.call.bind(obj);
};

const assert = require('assert');
const callBind = require('./index');
const callBound = require('./callBound');

// Test callBind
function f(a, b) {
  assert.equal(this, 1);
  assert.equal(a, 2);
  assert.equal(b, 3);
  assert.equal(arguments.length, 2);
}

const fBound = callBind(f);
delete Function.prototype.call;
delete Function.prototype.bind;

fBound(1, 2, 3);

// Test callBound
const slice = callBound('Array.prototype.slice');
assert.deepEqual(slice([1, 2, 3, 4], 1, -1), [2, 3]);
```
