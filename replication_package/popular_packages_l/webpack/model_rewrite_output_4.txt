// File: build.js
const fs = require('fs');
const path = require('path');
const babel = require('babel-core');

async function loadModule(modulePath) {
  const code = await fs.promises.readFile(modulePath, 'utf-8');
  console.log(`Loaded module: ${modulePath}`);
  return code;
}

async function bundle(entryModule) {
  const entryCode = await loadModule(entryModule);

  const transpiled = babel.transform(entryCode, { presets: ["env"] });
  console.log(`Transpiled module:\n${transpiled.code}`);
  
  await fs.promises.mkdir(path.join(__dirname, 'dist'), { recursive: true });
  await fs.promises.writeFile(path.join(__dirname, 'dist', 'bundle.js'), transpiled.code);

  console.log('Bundled code to dist/bundle.js');
}

bundle(path.join(__dirname, 'src', 'index.js')).catch(console.error);
