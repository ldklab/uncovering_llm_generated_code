1. **For Node.js 12 and above**: It checks if `process.allowedNodeEnvironmentFlags` exists and if it contains the `--preserve-symlinks` flag. This object `process.allowedNodeEnvironmentFlags` is a Set that contains all allowed CLI flags for the current Node.js version.

2. **For Node.js versions 6.2 to 11**: It checks if the `module.constructor._findPath` function (a private API - hence the use of an underscore) contains the string `preserveSymlinks`. This string would be present in Node.js versions that internally handle the `preserve-symlinks` option.

const isPreserveSymlinksSupported = (() => {
    // For Node.js 12 and above
    if (process.allowedNodeEnvironmentFlags &&
        process.allowedNodeEnvironmentFlags.has('--preserve-symlinks')) {
        return true;
    }

    // For Node.js versions 6.2 to 11
    if (String(module.constructor._findPath).includes('preserveSymlinks')) { // `includes` is simpler
        return true;
    }

    return false;
})();

module.exports = isPreserveSymlinksSupported;
