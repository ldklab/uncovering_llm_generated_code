const lazyExport = (exportName, loadFn) => {
	let cachedValue;
	Object.defineProperty(exports, exportName, {
		get() {
			if (loadFn !== undefined) {
				cachedValue = loadFn();
				loadFn = undefined;
			}
			return cachedValue;
		},
		configurable: true
	});
};
