The given Node.js code is a plugin for `figlet.js` that manages fonts used to generate ASCII art text. The code offers both asynchronous and synchronous methods for loading fonts into the `figlet` object and retrieving available font names:

1. **Asynchronous Font Loading (`figlet.loadFont`)**: 
   - Checks if the font is already loaded; if it is, it returns the cached version.
   - Reads the font file from the disk in UTF-8 encoding.
   - Parses the font data and adds it to `figlet`.
   - Supports a callback pattern for backward compatibility while also returning a Promise.

2. **Synchronous Font Loading (`figlet.loadFontSync`)**:
   - Similar to the asynchronous version but reads the files synchronously.
   - Returns the parsed font directly.

3. **Font Listing (`figlet.fonts`)**:
   - Reads all `.flf` files in the fonts directory to list available fonts.
   - Asynchronous version supports both callbacks and Promises.

4. **Synchronous Font Listing (`figlet.fontsSync`)**:
   - Synchronously lists all `.flf` files in the fonts directory.

Below is the rewritten version of the code:

```javascript
const figlet = require("./figlet.js");
const fs = require("fs");
const path = require("path");
const fontDir = path.join(__dirname, "/../fonts/");

figlet.loadFont = function (name, next) {
  return new Promise((resolve, reject) => {
    if (figlet.figFonts[name]) {
      next?.(null, figlet.figFonts[name].options);
      resolve(figlet.figFonts[name].options);
      return;
    }

    fs.readFile(path.join(fontDir, name + ".flf"), { encoding: "utf-8" }, (err, fontData) => {
      if (err) {
        next?.(err);
        reject(err);
        return;
      }

      fontData = fontData.toString(); // Ensures it's a string
      try {
        const font = figlet.parseFont(name, fontData);
        next?.(null, font);
        resolve(font);
      } catch (error) {
        next?.(error);
        reject(error);
      }
    });
  });
};

figlet.loadFontSync = function (name) {
  if (figlet.figFonts[name]) {
    return figlet.figFonts[name].options;
  }

  const fontData = fs.readFileSync(path.join(fontDir, name + ".flf"), { encoding: "utf-8" });
  return figlet.parseFont(name, fontData.toString());
};

figlet.fonts = function (next) {
  return new Promise((resolve, reject) => {
    fs.readdir(fontDir, (err, files) => {
      if (err) {
        next?.(err);
        reject(err);
        return;
      }

      const fontList = files.filter(file => file.endsWith(".flf")).map(file => file.replace(/\.flf$/, ""));
      next?.(null, fontList);
      resolve(fontList);
    });
  });
};

figlet.fontsSync = function () {
  return fs.readdirSync(fontDir)
           .filter(file => file.endsWith(".flf"))
           .map(file => file.replace(/\.flf$/, ""));
};

module.exports = figlet;
```
