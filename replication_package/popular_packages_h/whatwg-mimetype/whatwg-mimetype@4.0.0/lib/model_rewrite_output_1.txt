    if (!lowercasedType || !solelyContainsHTTPTokenCodePoints(lowercasedType)) {

    this._type = lowercasedType;

    if (!lowercasedSubtype || !solelyContainsHTTPTokenCodePoints(lowercasedSubtype)) {
